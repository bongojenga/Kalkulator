    <script>

        // Prośba użytkownika o wprowadzenie danych

        var x = prompt("Podaj rozmiar zmiany (mm):");

        var y = prompt("Podaj odległość od końca do środka strefy chłodzenia (mm):");


        // Obliczenia

        var z = y - (x / 2);


        // Wyświetlanie wyniku DO WYJEBANIA

        // document.write("Położenie igły wewnątrz zmiany: " + z + " (mm)");


        // Dodanie informacji dodatkowej na podstawie wartości z

        if (z > 0) {

            document.write("Wierzchołek kriosondy powinien wystawać poza guz " + Math.abs(z) + " (mm).");

        } else if (z < 0) {

            document.write("Wierzchołek kriosondy powinien pozostać wewnątrz guza " + Math.abs(z) + " (mm) od jego końca.");

        } else {

            document.write("Wierzchołek kriosondy powinien znaleźć się dokładnie na końcu guza.");

        }

    </script>
